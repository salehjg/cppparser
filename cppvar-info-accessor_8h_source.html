71inline std::uint8_t effectivePtrLevel(const CppConstVarTypeEPtr& varType)
72{
73 return ptrLevel(varType) + [&varType]() {
74 switch (refType(varType))
75 {
76 case CppRefType::kByRef:
77 case CppRefType::kRValRef:
78 return 1;
79 case CppRefType::kNoRef:
80 break;
81 }
82 return 0;
83 }();
84}

116inline bool isVoid(const CppVarType* varType)
117{
118 if (varType->typeModifier().ptrLevel_ != 0 || varType->typeModifier().refType_ != CppRefType::kNoRef)
119 return false;
120 // return (varType->baseType().compare("void") == 0);
121 // Above simple check fails to detect cases like usage of GrGLvoid
122 if (varType->baseType().length() < 4)
123 return false;
124 return (strncmp(varType->baseType().c_str() + varType->baseType().length() - 4, "void", 4) == 0);
125}
