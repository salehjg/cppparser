28struct CppTypeModifier
29{
30 CppRefType refType_;
31 std::uint8_t ptrLevel_; // Pointer level. e.g. int** ppi has pointer level of 2.
32
33 // Stores bits as per location of const in a var definition.
34 // Below table clarifies the value of const-bits:
35 // --------------------------------------------------
36 // | DEFINITION | VALUE |
37 // | ------------------------------------------------
38 // | int const * pi | 0b001 |
39 // | int const i | 0b001 |
40 // | int * const pi | 0b010 |
41 // | int * const * const ppi | 0b110 |
42 // | int **const ppi | 0b100 |
43 // | int const * const * const ppi | 0b111 |
44 // | ------------------------------------------------
45 //
46 // It is 8 bit unsigned integer which is enough to store info for pointers of 8 level deep.
47 std::uint32_t constBits_;
48};

50inline CppTypeModifier makeCppTypeModifier(CppRefType refType, std::uint8_t ptrLevel, std::uint8_t constBits)

52 return CppTypeModifier {refType, ptrLevel, constBits};
