109bool CppParser::addRenamedKeyword(const std::string& keyword, std::string renamedKeyword)
110{
111 extern int GetKeywordId(const std::string& keyword);
112 auto id = GetKeywordId(keyword);
113 if (id == -1)
114 return false;
115 gRenamedKeywords.emplace(std::make_pair(std::move(renamedKeyword), id));
116
117 return true;
118}

130CppCompoundPtr CppParser::parseFile(const std::string& filename)
131{
132 auto stm = readFile(filename);
133 auto cppCompound = parseStream(stm.data(), stm.size());
134 if (!cppCompound)
135 return cppCompound;
136 cppCompound->name(filename);
137 return cppCompound;
138}

141{
