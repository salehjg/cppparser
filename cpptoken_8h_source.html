33// For holding tokens
34struct CppToken
35{
36 const char* sz;
37 size_t len;
38
39 bool operator==(const CppToken& rhs) const
40 {
41 if (len != rhs.len)
42 return false;
43 return (sz == rhs.sz) || (std::strncmp(sz, rhs.sz, len) == 0);
44 }
45
46 bool operator!=(const CppToken& rhs) const
47 {
48 if (len != rhs.len)
49 return true;
50 return std::strncmp(sz, rhs.sz, len) != 0;
51 }
52
53 operator std::string() const
54 {
55 return toString();
56 }
57
58 std::string toString() const
59 {
60 return sz ? std::string(sz, len) : std::string();
61 }
62};

67inline CppToken makeCppToken(const char* sz, size_t len)
68{
69 CppToken tkn = {sz, len};
70 return tkn;
71}
72
73inline CppToken makeCppToken(const char* beg, const char* end)
74{
75 return makeCppToken(beg, end - beg);
76}
77
78inline CppToken mergeCppToken(const CppToken& token1, const CppToken& token2)
79{
80 if (token1.sz == nullptr)
81 return token2;
82 else if (token2.sz == nullptr)
83 return token1;
84 return makeCppToken(token1.sz, token2.sz + token2.len - token1.sz);
85}

87template <typename _ST>
88inline _ST& operator<<(_ST& stm, const CppToken& token)
89{
90 for (size_t i = 0; i < token.len; ++i)
91 stm << token.sz[i];
92 return stm;
93}
