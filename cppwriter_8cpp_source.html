35static void emitAttribute(std::uint32_t attr, std::ostream& stm)
36{
37 if (attr & kStatic)
38 stm << "static ";
39 else if (attr & kExtern)
40 stm << "extern ";
41 else if (attr & kExternC)
42 stm << "extern C ";
43
44 if (attr & kConst)
45 stm << "const ";
46 if (attr & kConstExpr)
47 stm << "constexpr ";
48 if (attr & kVolatile)
49 stm << "volatile ";
50 if (attr & kMutable)
51 stm << "mutable ";
52}

54static void emitTypeModifier(const CppTypeModifier& modifier, std::ostream& stm)
55{
56 std::uint8_t constBit = 0;
57 for (constBit = 0; constBit < modifier.ptrLevel_; ++constBit)
58 {
59 if (modifier.constBits_ & (1 << constBit))
60 stm << " const ";
61 stm << "*";
62 }
63 if (modifier.constBits_ & (1 << constBit))
64 stm << " const";
65 if (modifier.refType_ == CppRefType::kByRef)
66 stm << '&';
67 else if (modifier.refType_ == CppRefType::kRValRef)
68 stm << "&&";
69}

77void CppWriter::emit(const CppObj* cppObj, std::ostream& stm, CppIndent indentation, bool noNewLine) const
78{
79 switch (cppObj->objType_)
80 {
81 case CppObjType::kHashDefine:
82 return emitDefine((CppDefine*) cppObj, stm);
83 case CppObjType::kHashUndef:
84 return emitUndef((CppUndef*) cppObj, stm);
85 case CppObjType::kHashInclude:
86 return emitInclude((CppInclude*) cppObj, stm);
87 case CppObjType::kHashIf:
88 return emitHashIf((CppHashIf*) cppObj, stm);
89 case CppObjType::kHashPragma:
90 return emitPragma((CppPragma*) cppObj, stm);
91
92 case CppObjType::kVarType:
93 return emitVarType((CppVarType*) cppObj, stm);
94 case CppObjType::kVar:
95 {
96 emitVar((CppVar*) cppObj, stm, indentation);
97 if (!noNewLine)
98 stm << ";\n";
99 return;
100 }
101 case CppObjType::kVarList:
102 return emitVarList((CppVarList*) cppObj, stm, indentation);
103 case CppObjType::kEnum:
104 return emitEnum((CppEnum*) cppObj, stm, !noNewLine, indentation);
105 case CppObjType::kDocComment:
106 return emitDocComment((CppDocComment*) cppObj, stm, indentation);
107 case CppObjType::kUsingDecl:
108 return emitUsingDecl((CppUsingDecl*) cppObj, stm, indentation);
109 case CppObjType::kTypedefName:
110 return emitTypedef((CppTypedefName*) cppObj, stm, indentation);
111 case CppObjType::kTypedefNameList:
112 return emitTypedefList((CppTypedefList*) cppObj, stm, indentation);
113 case CppObjType::kCompound:
114 return emitCompound((CppCompound*) cppObj, stm, indentation, !noNewLine);
115 case CppObjType::kFwdClsDecl:
116 return emitFwdDecl((CppFwdClsDecl*) cppObj, stm, indentation);
117 case CppObjType::kFunction:
118 return emitFunction((CppFunction*) cppObj, stm, !noNewLine, indentation);
119 case CppObjType::kConstructor:
120 return emitConstructor((CppConstructor*) cppObj, stm, indentation);
121 case CppObjType::kDestructor:
122 return emitDestructor((CppDestructor*) cppObj, stm, indentation);
123 case CppObjType::kTypeConverter:
124 return emitTypeConverter((CppTypeConverter*) cppObj, stm, indentation);
125 case CppObjType::kFunctionPtr:
126 return emitFunctionPtr((CppFunctionPointer*) cppObj, stm, !noNewLine, indentation);
127 case CppObjType::kIfBlock:
128 return emitIfBlock((CppIfBlock*) cppObj, stm, indentation);
129 case CppObjType::kWhileBlock:
130 return emitWhileBlock((CppWhileBlock*) cppObj, stm, indentation);
131 case CppObjType::kDoWhileBlock:
132 return emitDoBlock((CppDoWhileBlock*) cppObj, stm, indentation);
133 case CppObjType::kForBlock:
134 return emitForBlock((CppForBlock*) cppObj, stm, indentation);
135 case CppObjType::kExpression:
136 emitExpr((CppExpr*) cppObj, stm, indentation);
137 if (!noNewLine)
138 stm << ";\n";
139 break;
140 case CppObjType::kSwitchBlock:
141 return emitSwitchBlock(static_cast<const CppSwitchBlock*>(cppObj), stm, indentation);
142 case CppObjType::kMacroCall:
143 return emitMacroCall(static_cast<const CppMacroCall*>(cppObj), stm, indentation);
144
145 case CppObjType::kBlob:
146 return emitBlob((CppBlob*) cppObj, stm, true, indentation);
147
148 default:
149 break;
150 }
151}

153void CppWriter::emitDefine(const CppDefine* defObj, std::ostream& stm) const
154{
155 stm << '#' << preproIndent_ << "define " << defObj->name_;
156 if (!defObj->defn_.empty())
157 {
158 const auto firstNonSpaceCharPos =
159 std::find_if(defObj->defn_.begin(), defObj->defn_.end(), [](char c) { return !std::isspace(c); });
160 if (firstNonSpaceCharPos != defObj->defn_.end())
161 {
162 if (*firstNonSpaceCharPos != '(')
163 stm << '\t';
164 stm << defObj->defn_;
165 }
166 }
167 stm << '\n';
168}

189void CppWriter::emitHashIf(CppHashIf::CondType condType, const std::string& cond, std::ostream& stm) const
190{
191 switch (condType)
192 {
193 case CppHashIf::kIf:
194 stm << '#' << preproIndent_ << "if " << cond << '\n';
195 ++preproIndent_;
196 break;
197
198 case CppHashIf::kIfDef:
199 stm << '#' << preproIndent_ << "ifdef " << cond << '\n';
200 ++preproIndent_;
201 break;
202
203 case CppHashIf::kIfNDef:
204 stm << '#' << preproIndent_ << "ifndef " << cond << '\n';
205 ++preproIndent_;
206 break;
207
208 case CppHashIf::kElIf:
209 stm << '#' << --preproIndent_ << "elif " << cond << '\n';
210 ++preproIndent_;
211 break;
212
213 case CppHashIf::kElse:
214 stm << '#' << --preproIndent_ << "else " << cond << '\n';
215 ++preproIndent_;
216 break;
217
218 case CppHashIf::kEndIf:
219 emitEndIf(stm);
220 break;
221 }
222}

254void CppWriter::emitVarType(const CppVarType* varTypeObj, std::ostream& stm) const
255{
256 const auto attr = varTypeObj->typeAttr() | (isConst(varTypeObj) ? CppIdentifierAttrib::kConst : 0);
257 emitAttribute(attr, stm);
258 if (varTypeObj->compound())
259 emit(varTypeObj->compound(), stm, CppIndent(), true);
260 else
261 stm << varTypeObj->baseType();
262 const auto& origTypeModifier = varTypeObj->typeModifier();
263 const CppTypeModifier typeModifier {
264 origTypeModifier.refType_, origTypeModifier.ptrLevel_, origTypeModifier.constBits_ & ~1};
265 emitTypeModifier(typeModifier, stm);
266 if (varTypeObj->paramPack_)
267 stm << "...";
268}

277void CppWriter::emitVarDecl(std::ostream& stm, const CppVarDecl& varDecl, bool skipName) const
278{
279 if (!skipName && !varDecl.name().empty())
280 stm << varDecl.name();
281 for (const auto& arrSize : varDecl.arraySizes())
282 {
283 stm << '[';
284 if (arrSize)
285 emitExpr(arrSize.get(), stm);
286 stm << ']';
287 }
288 if (varDecl.assignType() == AssignType::kUsingEqual)
289 {
290 stm << " = ";
291 emit(varDecl.assignValue(), stm, CppIndent(), true);
292 }
293 else if (varDecl.assignType() == AssignType::kUsingBracket)
294 {
295 stm << '(';
296 if (varDecl.assignValue())
297 emit(varDecl.assignValue(), stm, CppIndent(), true);
298 stm << ')';
299 }
300 else if (varDecl.assignType() == AssignType::kUsingBraces)
301 {
302 stm << '{';
303 if (varDecl.assignValue())
304 emit(varDecl.assignValue(), stm, CppIndent(), true);
305 stm << '}';
306 }
307}

309void CppWriter::emitVar(const CppVar* varObj, std::ostream& stm, CppIndent indentation, bool skipName) const
310{
311 stm << indentation;
312 if (!varObj->apidecor().empty())
313 {
314 stm << varObj->apidecor() << ' ';
315 }
316 emitVarType(varObj->varType(), stm);
317 if (!skipName && !varObj->name().empty())
318 stm << ' ';
319 emitVarDecl(stm, varObj->varDecl(), skipName);
320}

322void CppWriter::emitVarList(const CppVarList* varListObj,
323 std::ostream& stm,
324 CppIndent indentation /* = CppIndent()*/) const
325{
326 emitVar(varListObj->firstVar().get(), stm, indentation);
327 auto& varDeclList = varListObj->varDeclList();
328 for (size_t i = 0; i < varDeclList.size(); ++i)
329 {
330 stm << ", ";
331 const auto& decl = varDeclList[i];
332 emitTypeModifier(decl, stm);
333 emitVarDecl(stm, decl, false);
334 }
335
336 stm << ";\n";
337}

339void CppWriter::emitEnum(const CppEnum* enmObj, std::ostream& stm, bool emitNewLine, CppIndent indentation) const
340{
341 stm << indentation << "enum";
342 if (enmObj->isClass_)
343 stm << " class";
344 if (!enmObj->name_.empty())
345 stm << ' ' << enmObj->name_;
346 if (!enmObj->underlyingType_.empty())
347 stm << " : " << enmObj->underlyingType_;
348 if (enmObj->itemList_)
349 {
350 const bool isEnumBodyBlob = !enmObj->itemList_->empty() && enmObj->itemList_->front()->val_
351 && (enmObj->itemList_->front()->val_->objType_ == CppBlob::kObjectType);
352 if (isEnumBodyBlob)
353 {
354 stm << " {\n";
355 emitBlob((CppBlob*) enmObj->itemList_->front()->val_.get(), stm, false, indentation);
356 stm << '\n' << indentation << '}';
357 }
358 else
359 {
360 stm << '\n';
361 stm << indentation++ << "{\n";
362 for (auto enmItem : *(enmObj->itemList_))
363 {
364 if (enmItem->name_.empty())
365 {
366 emit(enmItem->val_.get(), stm, indentation);
367 }
368 else
369 {
370 stm << indentation << enmItem->name_;
371 if (enmItem->val_ && isExpr(enmItem->val_.get()))
372 {
373 auto* expr = static_cast<CppExpr*>(enmItem->val_.get());
374 stm << " = ";
375 emitExpr(expr, stm);
376 }
377 if (enmItem != enmObj->itemList_->back())
378 stm << ",\n";
379 else
380 stm << '\n';
381 }
382 }
383 stm << --indentation << "}";
384 }
385 }
386 if (emitNewLine)
387 stm << ";\n";
388}

445void CppWriter::emitTemplSpec(const CppTemplateParamList* templSpec, std::ostream& stm, CppIndent indentation) const
446{
447 stm << indentation << "template <";
448 if (templSpec)
449 {
450 const char* sep = "";
451 for (auto& param : *templSpec)
452 {
453 stm << sep;
454 if (param->paramType_)
455 {
456 if (param->paramType_->objType_ == CppVarType::kObjectType)
457 emitVarType(static_cast<const CppVarType*>(param->paramType_.get()), stm);
458 else
459 emitFunctionPtr(static_cast<const CppFunctionPointer*>(param->paramType_.get()), stm, false);
460 stm << ' ';
461 }
462 else
463 {
464 stm << "typename ";
465 }
466 stm << param->paramName_;
467 if (param->defaultArg())
468 {
469 stm << " = ";
470 emit(param->defaultArg(), stm, CppIndent(), true);
471 }
472 sep = ", ";
473 }
474 }
475 stm << ">\n";
476}

478void CppWriter::emitCompound(const CppCompound* compoundObj,
479 std::ostream& stm,
480 CppIndent indentation,
481 bool emitNewLine) const
482{
483 if (isNamespaceLike(compoundObj))
484 {
485 if (compoundObj->templateParamList())
486 {
487 emitTemplSpec(compoundObj->templateParamList(), stm, indentation);
488 }
489 stm << indentation << compoundObj->compoundType() << ' ';
490 if (!compoundObj->apidecor().empty())
491 stm << compoundObj->apidecor() << ' ';
492 stm << compoundObj->name();
493 if (compoundObj->hasAttr(kFinal))
494 stm << " final";
495 }
496 if (compoundObj->inheritanceList())
497 {
498 ++indentation;
499 char sep = ':';
500 stm << ' ';
501 for (CppInheritanceList::const_iterator inhItr = compoundObj->inheritanceList()->begin();
502 inhItr != compoundObj->inheritanceList()->end();
503 ++inhItr)
504 {
505 stm << sep << ' ' << inhItr->inhType << ' ' << inhItr->baseName;
506 sep = ',';
507 }
508 --indentation;
509 }
510 if (isNamespaceLike(compoundObj))
511 stm << '\n' << indentation++ << "{\n";
512 else if (compoundObj->compoundType() == CppCompoundType::kExternCBlock)
513 stm << indentation++ << "extern \"C\" {\n";
514
515 CppAccessType lastAccessType = CppAccessType::kUnknown;
516 forEachMember(compoundObj, [&](const CppObj* memObj) {
517 if (isClassLike(compoundObj) && memObj->accessType_ != CppAccessType::kUnknown
518 && lastAccessType != memObj->accessType_)
519 {
520 stm << --indentation << memObj->accessType_ << ':' << '\n';
521 lastAccessType = memObj->accessType_;
522 ++indentation;
523 }
524 emit(memObj, stm, indentation);
525 if (memObj->objType_ == CppBlob::kObjectType)
526 stm << '\n';
527
528 return false;
529 });
530
531 if (isNamespaceLike(compoundObj))
532 {
533 stm << --indentation;
534 stm << '}';
535 if (emitNewLine)
536 {
537 if (isClassLike(compoundObj))
538 stm << ';';
539 stm << '\n';
540 }
541 }
542 else if (compoundObj->compoundType() == CppCompoundType::kExternCBlock)
543 stm << indentation << "}\n";
544}

551void CppWriter::emitParamList(const CppParamVector* paramListObj, std::ostream& stm, bool skipParamName) const
552{
553 for (auto prmItr = paramListObj->begin(); prmItr != paramListObj->end(); ++prmItr)
554 {
555 if (prmItr != paramListObj->begin())
556 stm << ", ";
557 auto& param = *prmItr;
558 switch (param->objType_)
559 {
560 case CppObjType::kVar:
561 emitVar(static_cast<const CppVar*>(param.get()), stm, CppIndent(), skipParamName);
562 break;
563 case CppObjType::kFunctionPtr:
564 emitFunctionPtr(static_cast<const CppFunctionPointer*>(param.get()), stm, skipParamName);
565 break;
566 default:
567 assert(false);
568 }
569 }

572void CppWriter::emitFunction(const CppFunction* funcObj,
573 std::ostream& stm,
574 CppIndent indentation,
575 bool skipName,
576 bool skipParamName,
577 bool emitNewLine) const
578{
579 if (funcObj->templateParamList())
580 emitTemplSpec(funcObj->templateParamList(), stm, indentation);
581
582 if ((funcObj->attr() & (kFuncParam | kTypedef)) == 0)
583 stm << indentation;
584 if (!funcObj->decor1().empty())
585 stm << funcObj->decor1() << ' ';
586 if (funcObj->hasAttr(kStatic))
587 stm << "static ";
588 else if (funcObj->hasAttr(kExtern))
589 stm << "extern ";
590 else if (funcObj->hasAttr(kVirtual) && !(funcObj->hasAttr(kOverride) || funcObj->hasAttr(kFinal)))
591 stm << "virtual ";
592 else if (funcObj->hasAttr(kInline))
593 stm << "inline ";
594 else if (funcObj->hasAttr(kExplicit))
595 stm << "explicit ";
596 else if (funcObj->hasAttr(kFriend))
597 stm << "friend ";
598 if (funcObj->hasAttr(kConstExpr))
599 stm << "constexpr ";
600 if (funcObj->hasAttr(kTrailingRet))
601 stm << "auto";
602 else
603 emitVarType(funcObj->retType_.get(), stm);
604 if (funcObj->objType_ == CppObjType::kFunctionPtr)
605 stm << " (";
606 else
607 stm << ' ';
608 if (!funcObj->decor2().empty())
609 stm << funcObj->decor2() << ' ';
610 if (funcObj->objType_ == CppObjType::kFunctionPtr)
611 {
612 stm << '*';
613 if (!skipName)
614 stm << funcObj->name_ << ") ";
615 }
616 else if (!skipName)
617 {
618 stm << funcObj->name_;
619 }
620 stm << '(';
621 if (funcObj->params())
622 emitParamList(funcObj->params(), stm, skipParamName);
623 stm << ')';
624
625 if ((funcObj->attr() & kConst) == kConst)
626 stm << " const";
627 if ((funcObj->attr() & kPureVirtual) == kPureVirtual)
628 stm << " = 0";
629 else if ((funcObj->attr() & kOverride) == kOverride)
630 stm << " override";
631 else if ((funcObj->attr() & kFinal) == kFinal)
632 stm << " final";
633
634 if (funcObj->attr() & kTrailingRet)
635 {
636 stm << " -> ";
637 emitVarType(funcObj->retType_.get(), stm);
638 }
639 if (!skipParamName && funcObj->defn() && (getEmittingType() != kHeader))
640 {
641 const auto defn = funcObj->defn();
642 if (defn->hasASingleBlobMember())
643 {
644 stm << '\n' << indentation++ << "{\n";
645 emitBlob((CppBlob*) defn->members().front().get(), stm, false, indentation);
646 stm << '\n' << --indentation << "}\n";
647 }
648 else
649 {
650 stm << '\n' << indentation++ << "{\n";
651 emitCompound(funcObj->defn(), stm, indentation);
652 stm << --indentation << "}\n";
653 }
654 }
655 else if (emitNewLine && ((funcObj->attr() & kFuncParam) == 0))
656 {
657 stm << ";\n";
658 }
659}

679void CppWriter::emitConstructor(const CppConstructor* ctorObj,
680 std::ostream& stm,
681 CppIndent indentation,
682 bool skipParamName) const
683{
684 if (ctorObj->templateParamList())
685 {
686 emitTemplSpec(ctorObj->templateParamList(), stm, indentation);
687 }
688 stm << indentation;
689 if (!ctorObj->decor1().empty())

961void CppWriter::emitExpr(const CppExpr* exprObj, std::ostream& stm, CppIndent indentation /*= CppIndent()*/) const
962{
963 if (exprObj == NULL)
964 return;
965 stm << indentation;
966 if (exprObj->flags_ & CppExpr::kReturn)
967 stm << "return ";
968 if (exprObj->flags_ & CppExpr::kThrow)
969 stm << "throw ";
970 if (exprObj->flags_ & CppExpr::kInitializer)
971 stm << "{";
972 if (exprObj->flags_ & CppExpr::kBracketed)
973 stm << '(';
974 if (exprObj->flags_ & CppExpr::kNew)
975 stm << "new ";
976 if (exprObj->flags_ & CppExpr::kSizeOf)
977 stm << "sizeof(";
978 else if (exprObj->flags_ & CppExpr::kDelete)
979 stm << "delete ";
980 else if (exprObj->flags_ & CppExpr::kDeleteArray)
981 stm << "delete[] ";
982 if (exprObj->oper_ == kNone)
983 {
984 emitExprAtom(exprObj->expr1_, stm);
985 }
986 else if (exprObj->oper_ > kUnariPrefixOperatorStart && exprObj->oper_ < kUnariSufixOperatorStart)
987 {
988 emitOperator(stm, exprObj->oper_);
989 emitExprAtom(exprObj->expr1_, stm);
990 }
991 else if (exprObj->oper_ > kUnariSufixOperatorStart && exprObj->oper_ < kBinaryOperatorStart)
992 {
993 emitExprAtom(exprObj->expr1_, stm);
994 emitOperator(stm, exprObj->oper_);
995 }
996 else if (exprObj->oper_ > kBinaryOperatorStart && exprObj->oper_ < kDerefOperatorStart)
997 {
998 emitExprAtom(exprObj->expr1_, stm);
999 if (exprObj->oper_ != kComma)
1000 stm << ' ';
1001 emitOperator(stm, exprObj->oper_);
1002 stm << ' ';
1003 emitExprAtom(exprObj->expr2_, stm);
1004 }
1005 else if (exprObj->oper_ > kDerefOperatorStart && exprObj->oper_ < kSpecialOperations)
1006 {
1007 emitExprAtom(exprObj->expr1_, stm);
1008 emitOperator(stm, exprObj->oper_);
1009 emitExprAtom(exprObj->expr2_, stm);
1010 }
1011 else if (exprObj->oper_ == kFunctionCall)
1012 {
1013 emitExprAtom(exprObj->expr1_, stm);
1014 stm << '(';
1015 emitExprAtom(exprObj->expr2_, stm);
1016 stm << ')';
1017 }
1018 else if (exprObj->oper_ == kUniformInitCall)
1019 {
1020 emitExprAtom(exprObj->expr1_, stm);
1021 stm << '{';
1022 emitExprAtom(exprObj->expr2_, stm);
1023 stm << '}';
1024 }
1025 else if (exprObj->oper_ == kArrayElem)
1026 {
1027 emitExprAtom(exprObj->expr1_, stm);
1028 stm << '[';
1029 emitExprAtom(exprObj->expr2_, stm);
1030 stm << ']';
1031 }
1032 else if (exprObj->oper_ == kCStyleCast)
1033 {
1034 stm << '(';
1035 emitExprAtom(exprObj->expr1_, stm);
1036 stm << ") ";
1037 emitExprAtom(exprObj->expr2_, stm);
1038 }
1039 else if (exprObj->oper_ >= kConstCast && exprObj->oper_ <= kReinterpretCast)
1040 {
1041 if (exprObj->oper_ == kConstCast)
1042 stm << "const_cast";
1043 else if (exprObj->oper_ == kStaticCast)
1044 stm << "static_cast";
1045 else if (exprObj->oper_ == kDynamicCast)
1046 stm << "dynamic_cast";
1047 else if (exprObj->oper_ == kReinterpretCast)
1048 stm << "reinterpret_cast";
1049 stm << '<';
1050 emitExprAtom(exprObj->expr1_, stm);
1051 stm << ">(";
1052 emitExprAtom(exprObj->expr2_, stm);
1053 stm << ')';
1054 }
1055 else if (exprObj->oper_ == kTertiaryOperator)
1056 {
1057 emitExprAtom(exprObj->expr1_, stm);
1058 stm << " ? ";
1059 emitExprAtom(exprObj->expr2_, stm);
1060 stm << " : ";
1061 emitExprAtom(exprObj->expr3_, stm);
1062 }
1063 else if (exprObj->oper_ == kPlacementNew)
1064 {
1065 stm << "new (";
1066 emitExprAtom(exprObj->expr1_, stm);
1067 stm << ") ";
1068 emitExprAtom(exprObj->expr2_, stm);
1069 }
1070
1071 if (exprObj->flags_ & CppExpr::kBracketed)
1072 stm << ')';
1073 if (exprObj->flags_ & CppExpr::kInitializer)
1074 stm << "}";
1075 if (exprObj->flags_ & CppExpr::kSizeOf)
1076 stm << ')';
1077
1078 if (exprObj->flags_ & CppExpr::kVariadicPack)
1079 stm << "...";
1080}
